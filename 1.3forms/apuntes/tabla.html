Atributo Valor Descripción Ejemplo
accept Tipo de MIME Especifica los tipos de archivos que se pueden seleccionar en un campo de tipo "file" accept=".jpg,.png,.gif"
alt Texto alternativo Texto que se muestra cuando la imagen no se puede cargar alt="Descripción de la imagen"
autocomplete on Especifica si el navegador debe autocompletar el campo autocomplete="on"
checked checked Indica que la casilla de verificación está seleccionada por defecto <input type="checkbox" checked>
cols Número Número de columnas en un campo de tipo "textarea" cols="40"
disabled disabled Deshabilita el control <input type="text" disabled>
enctype application/x-www-form-urlencoded Tipo de codificación de los datos del formulario enctype="application/x-www-form-urlencoded"
for ID del elemento Asocia la etiqueta con un elemento de formulario for="nombre"
height Número Altura de un campo de tipo "textarea" height="150px"
id ID único Identificador único para el elemento id="miFormulario"
list ID del elemento Asocia un elemento "datalist" con el campo de entrada list="sugerencias"
max Valor máximo Valor máximo que se puede introducir en un campo de tipo "number" max="100"
maxlength Número Número máximo de caracteres que se pueden introducir en un campo maxlength="20"
min Valor mínimo Valor mínimo que se puede introducir en un campo de tipo "number" min="1"
multiple multiple Permite seleccionar múltiples opciones en un campo de tipo "select" <select multiple>
name Nombre del campo Nombre del campo que se enviará al servidor name="nombre"
pattern Expresión regular Patrón de caracteres que se deben introducir en un campo pattern="[A-Za-z0-9_-]+"
placeholder Texto Texto que se muestra en un campo vacío como sugerencia placeholder="Introduce tu nombre"
readonly readonly Hace que el campo sea de solo lectura <input type="text" readonly>
required required Indica que el campo es obligatorio <input type="text" required>
rows Número Número de filas en un campo de tipo "textarea" rows="10"
size Número Tamaño del campo de entrada size="20"
src URL URL de la imagen que se muestra en un campo de tipo "image" src="imagen.jpg"
step Valor Incremento o decremento permitido en un campo de tipo "number" step="0.1"
target _blank Ventana en la que se abrirá el enlace al enviar el formulario target="_blank"
type Tipo de campo Especifica el tipo de campo de formulario (text, password, checkbox, radio, etc.) type="text"
value Valor por defecto Valor inicial del campo value="correo@ejemplo.com"
width Número Ancho del campo de entrada width="100"
